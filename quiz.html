function generateQuizAI() { const subject = document.getElementById("quiz-subject").value; const topic = document.getElementById("quiz-topic").value.trim() || "Umumiy"; const count = document.getElementById("quiz-count").value; const statusDiv = document.getElementById("teacher-status"); statusDiv.innerHTML = ` Gemini AI ${subject} bo'yicha test yaratmoqda...`; const prompt = `Yaratish professional test tizimi uchun. Fan: ${subject}. Mavzu: ${topic}. Savollar soni: ${count}. Javobni faqat va faqat quyidagi JSON formatda qaytar, boshqa hech qanday matn qo'shma, \`\`\`json belgisini ham qo'shma: {"questions": [{"question": "Savol matni?", "a": "A varianti", "b": "B varianti", "c": "C varianti", "d": "D varianti", "answer": "a", "topic": "${topic}"}]}`; // Передаем токен через заголовки x-goog-api-key, чтобы точка в ключе 'AQ.' не ломала URL fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent`, { method: "POST", headers: { "Content-Type": "application/json", "x-goog-api-key": TOKEN }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }) .then(res => { if(!res.ok) throw new Error("API ulanish xatosi " + res.status); return res.json(); }) .then(data => { let aiText = data.candidates[0].content.parts[0].text.trim(); aiText = aiText.replace(/^```json/, "").replace(/```$/, "").trim(); const parsedData = JSON.parse(aiText); statusDiv.innerText = "Yaratildi! Google Jadvalga saqlanmoqda..."; return fetch(API_URL, { method: "POST", body: JSON.stringify({ type: "save_questions", subject: subject, questions: parsedData.questions }) }); }) .then(() => { statusDiv.innerHTML = ` Muvaffaqiyatli saqlandi!`; }) .catch(err => { console.error(err); statusDiv.innerHTML = ` Xatolik yuz berdi. Ключ API или сеть отклонили запрос.`; }); }
